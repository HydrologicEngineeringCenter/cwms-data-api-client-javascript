Example: Levels

An example for calling the Levels endpoint via CDA in JavaScript

Swagger UI for Endpoints

import { LevelsApi, Configuration } from "cwmsjs";


const l_api = new LevelsApi()
await l_api.getCwmsDataLevels({
    levelIdMask: "*KEYS*",
    office: "SWT",
    format: "json"
})
    .then((data) => {
        console.log(data?.['location-levels']?.['location-levels'])
        data?.['location-levels']?.['location-levels'].forEach((level) => {
            console.log(level?.name)
            console.log(level?.office)
            console.log(level?.values)
        })
    }).catch(async e => {
        if (e.response) {
            const error_msg = await e.response.json()
            e.message = `${e.response.url}\n${e.message}\n${JSON.stringify(error_msg, null, 2)}`;
            console.error(e);
        }
        throw e;
    })
